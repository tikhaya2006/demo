using System.Threading.Tasks;
using Npgsql;
using Magazin.Model;

namespace Magazin.Services;

public static class Database
{
    static string Conn = "Host=localhost;Port=5433;Database=demo;Username=postgres;Password=1234";
    public static async Task<User?> GetUserAsync(string login, string pass)
    {
        try
        {
            using var con = new NpgsqlConnection(Conn);
            await con.OpenAsync();
            using var cmd = new NpgsqlCommand
           (@"SELECT 
            u.id, 
            u.user_role_id, 
            r.name 
            FROM users u 
            JOIN user_role r ON u.user_role_id = r.id 
            WHERE u.login = @login 
            AND u.user_password = @pass", con);
            cmd.Parameters.AddWithValue("@login", login);
            cmd.Parameters.AddWithValue("@pass", pass);
            using var r = await cmd.ExecuteReaderAsync();
            return await r.ReadAsync() ? new User { id = r.GetInt32(0), RoleId = r.GetInt32(1), RoleName = r.GetString(2), Login = login } : null;
        }
        catch { return null; }
    }
}

using Avalonia.Controls;
using System;
using System.Threading.Tasks;
using Magazin.Services;
using Magazin.Model;

namespace Magazin.Views;

public partial class MainWindow : Window
{
    public MainWindow()
    {
        InitializeComponent();
        LoginBtn.Click += async (s, e) => await Login();
        GuestBtn.Click += (s, e) => GuestLogin();
    }
    private async Task Login()
    {
        var user = await Database.GetUserAsync(LoginBox.Text ?? "", PassBox.Text ?? "");
        
        if (user != null)
        {
            if (user.RoleId == 1)
                new AdminWindow(user).Show();
            else if (user.RoleId == 2)
                new ManagerWindow(user).Show();
            else
                new ClientWindow(user).Show();
            Close();
        }
        else
        {
            var error = new Window 
            { 
                Title = "Ошибка", 
                Width = 250, 
                Height = 100,
                Content = new TextBlock { Text = "Неверный логин или пароль", Margin = new(20) }
            };
            await error.ShowDialog(this);
        }
    }
    private void GuestLogin()
    {
        new MainWindow().Show();
        Close();
    }
}

using Avalonia;
using Avalonia.Controls.ApplicationLifetimes;
using Avalonia.Markup.Xaml;
using Magazin.Views;

namespace Magazin;

public partial class App : Application
{
    public override void Initialize()
    {
        AvaloniaXamlLoader.Load(this);
    }

    public override void OnFrameworkInitializationCompleted()
    {
        if (ApplicationLifetime is IClassicDesktopStyleApplicationLifetime desktop)
        {
            desktop.MainWindow = new MainWindow();
        }

        base.OnFrameworkInitializationCompleted();
    }
